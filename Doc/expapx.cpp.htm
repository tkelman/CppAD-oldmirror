# include <cmath>             // for fabs function
# include "ExpApx.hpp"        // definition of ExpApx algorithm
bool ExpApx(void)
{    double x     = .5;
     double e     = .2;
     double check = 1 + .5 + .125; // include 1 and only 1 term less than e
     bool   ok    = std::fabs( ExpApx(x, e) - check ) <= 1e-10; 
     return ok;
}
