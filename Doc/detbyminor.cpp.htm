Determinant Using Expansion by Minors: Example and Test


# include <CppAD/CppAD.h>
# include "DetByMinor.h"

bool DetByMinor()
{    bool ok = true;

     // dimension of the matrix
     size_t n = 3;

     // construct the determinat object
     CppAD::DetByMinor<double> Det(n);

     double  a[] = {
          1., 2., 3.,  // a[0] a[1] a[2]
          3., 2., 1.,  // a[3] a[4] a[5]
          2., 1., 2.   // a[6] a[7] a[8]
     };
     CppADvector<double> A(9);
     size_t i;
     for(i = 0; i < 9; i++)
          A[i] = a[i];


     // evaluate the determinant
     double det = Det(A);

     double check;
     check = a[0]*(a[4]*a[8] - a[5]*a[7])
           - a[1]*(a[3]*a[8] - a[5]*a[6])
           + a[2]*(a[3]*a[7] - a[4]*a[6]);

     ok = det == check;

     return ok;
}
